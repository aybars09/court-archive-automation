create database arsiv;
use arsiv;


-- 1 GRUP TABLOSU
-----------------------------------
create table groupss (
	id int AUTO_INCREMENT not null,	
	grupName  varchar(30) UNIQUE not null,
    primary key(id)
);




-- 2 PERSONEL  TABLOSU
----------------------------------
create table personel (
	id int AUTO_INCREMENT not null,	
	tcno varchar(11) unique,
	adsoyad varchar(30) not null,
	dtarihi varchar(30),
	calistigibirim varchar(30),
    primary key (id)
);



-- 3 USER TABLOSU
----------------------------------
create table USERS (
	id int AUTO_INCREMENT not null,		 
	userName  varchar(50) not null ,					 
	password  varchar(100),
	groupsID int not null,
	aitOldKisiID int unique not null,
    foreign key (groupsID) references  groupss(id),
    foreign key (aitOldKisiID) references  personel(id),
    primary key (id)
);





-- 4 MAHKEME TURU TABLOSU
----------------------------------
create table mahkemeturu (
	id int AUTO_INCREMENT not null primary key,			 
	mahkemeTuru  varchar(50) not null					 
);





-- 5 MAHKEMELER TABLOSU
----------------------------------
create table mahkemeler (
	id int AUTO_INCREMENT not null,			 
	adi  varchar(50) not null,
	turuID int,
    primary key (id),
    foreign key (turuID) references mahkemeturu(id)
);





-- 6 DAVA TURU TABLOSU
----------------------------------
create table davaturu (
	id int AUTO_INCREMENT not null,	
	imhaSuresi int,
	mahkemeID  int,
	davaninTuru varchar(100),
    primary key (id),
    foreign key (mahkemeID) references mahkemeler(id)
);






-- 7 ARSIV TABLOSU
----------------------------------
create table arsiv (
	id int AUTO_INCREMENT not null,	
	yil varchar(50),
    geldigiYerId int,
	dosyaEsasNo varchar(50),
	kararNo varchar(50),
	karartarihi varchar(50),
	kesinlesmeTarihi varchar(50),
	davaTuruID int,
    primary key (id),
    foreign key (geldigiYerId) references mahkemeler(id),
    foreign key (davaTuruID) references davaTuru(id)
);




-- 8 IMHA DA GOREVLI PERSONEL TABLOSU
----------------------------------
create table imha_personel (	
	id int AUTO_INCREMENT not null,	
    arsivId int ,
	persId int,
	imhaTarihi  varchar(50),
	primary key (id),
	foreign key (persId) references personel(id),
	foreign key (arsivId) references arsiv(id)
);
